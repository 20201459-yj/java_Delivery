create table account(
    ac_id varchar(40) unique, -- > ID속성
    ac_pw varchar(40), -- > PW속성
    ac_uid varchar(40) primary key -- > 고유식별번호, 고객은 C0식, 점주는 S0식, 어드민은 자바에서만 처리 / 이 값으로 로그인 후 회원 정보 접근
);

create table customer(
    cs_uid varchar(40) primary key, -- > 자바에 의해 C로 시작하는 값만 삽입됨, 자바에서 int가 아닌 String 타입으로 접근해야 하는 것 인지
    cs_name varchar(40),
    cs_address varchar(100), -- > 이 값에는 검증된 데이터(주소 실효성 확인된)만 삽입됨
    cs_phone varchar(40),
    cs_balance int, -- > 자바에서 초기 생성 시 0
    
    constraint foreign key(cs_uid) references account(ac_uid) on delete cascade
);

create table store(
    st_uid varchar(40) primary key, -- > 고객과 동일, 자바에 의해 S로 시작하는 값만 삽입
    st_name varchar(100),
    st_address varchar(100),
    st_phone varchar(40),
    st_type varchar(40), -- > 자바에 의해 한식, 중식, 일식, 양식, 카페 중 하나의 값만 삽입
    st_profit int, -- > 자바에서 초기 생성 시 0
    
    constraint foreign key(st_uid) references account(ac_uid) on delete cascade
);

create table menu(
    menu_no int auto_increment primary key,
    menu_store varchar(40), -- > 메뉴를 등록한 상점의 UID, 이 값을 활용하여 상점이 
    menu_name varchar(100),
    menu_price int,

    constraint foreign key(menu_store) references store(st_uid) on delete cascade
);

create table orders(
    od_no int auto_increment primary key,
    od_customer varchar(40), -- > cs_uid를 받음
    od_store varchar(40), -- > st_uid을 받음
    od_content varchar(40), -- > 주문 시 요청 사항
    od_menu int, -- > 주문한 메뉴, menu_no를 받음
    od_state int default 0, -- > 고객이 주문 시 0으로 인서트, 상점이 수락 시 1로 업데이트, 배달 완료 시 소멸(시간 남으면 2로 변경되며 주문이력)
    od_start datetime,
	od_duration int,

    constraint foreign key(od_customer) references customer(cs_uid) on delete cascade,
    constraint foreign key(od_store) references store(st_uid) on delete cascade,
    constraint foreign key(od_menu) references menu(menu_no) on delete cascade
);

create table review(
    rv_no int auto_increment primary key,
    rv_customer varchar(40), -- > cs_uid를 받음
    rv_menu int, -- > menu_no를 받음
    rv_score int, -- > 별점을 담당하며, 자바에 의해 1~5점의 값을 받음
    rv_content varchar(200), -- > 리뷰 내용

    constraint foreign key(rv_menu) references menu(menu_no) on delete cascade,
    constraint foreign key(rv_customer) references customer(cs_uid) on delete cascade
);
